\chapter{Orientation}

\section{The pipeline this blueprint documents}

The verified crate \texttt{rust/opendp\_verified} is processed by a fixed,
version-pinned pipeline:
\[
\text{\texttt{src/**/*.rs}} \xrightarrow{\;\text{Charon}\;} \text{\texttt{opendp\_verified.llbc}}
\xrightarrow{\;\text{Aeneas (-backend lean)}\;} \text{\texttt{Generated/OpenDP/*.lean}}.
\]
Aeneas extracts Rust into the \emph{deterministic} \(\Result\) monad. All
randomness enters through a single stochastic external, \fillbytes{} (OpenSSL).
Handwritten proofs in \texttt{src/**/*.lean} import the generated code and prove
each extracted function realises a SampCert reference distribution.

The development is layered: a shared \emph{semantic core}
(\texttt{src/core/primitives/}, Chapter~\ref{chap:foundation}) assigning a
sub-probability law to every \(\Result\) program and resting on a single
randomness axiom; a \emph{trust boundary} of deterministic external specs
(\texttt{src/core/externals/}, Chapter~\ref{chap:externals}); and one chapter per
sampler (\texttt{src/samplers/}) proving \(\sd(\text{extracted fn}) =
\text{target PMF}\). The core is what every sampler reuses --- the green spine of
the graph.

\section{How to read the status badges}

\begin{itemize}
  \item \statusDone{} --- statement and proof complete in Lean; green.
  \item \statusGap{} --- stated in Lean but the proof is not yet complete.
  \item \statusAxiom{} --- a declared \texttt{axiom}: trusted, not proved.
    \texttt{\#print axioms} on any end-to-end theorem enumerates the axioms it
    actually depends on.
  \item \statusStub{} --- no Lean statement yet; records the intended result so
    the frontier is visible (e.g.\ samplers not yet ported, or Rust not yet
    extracted into Lean at all).
\end{itemize}

\chapter{The probabilistic semantics (shared core)}\label{chap:foundation}
% =============================================================================

Files: \texttt{src/core/primitives/\{bytes,semantics\}.lean}. Every sampler proof
in later chapters bottoms out here.

\begin{definition}[Big-endian byte decoding]\label{def:beBytesToNat}
  \lean{OpenDP.Core.Bytes.beBytesToNat}\leanok
  \(\mathrm{beBytesToNat}\) interprets a byte list as a big-endian natural
  number. \statusDone
\end{definition}

\begin{definition}[Uniform byte distribution]\label{def:uniformByteNatPMF}
  \lean{OpenDP.Core.Bytes.uniformByteNatPMF}\leanok
  \uses{def:beBytesToNat}
  \(\mathrm{uniformByteNatPMF}\;n\) is the uniform law on \([0, 256^n)\): the
  target distribution of \(n\) freshly sampled bytes. The hardware-facing
  reference. \statusDone
\end{definition}

\begin{theorem}[Randomness axiom: a consistent semantics exists]\label{ax:sdg-exists}
  \lean{OpenDP.Core.Semantics.samplerDistGen_exists}\leanok
  \uses{def:uniformByteNatPMF}
  \statusAxiom{} There exists \(f : \forall A,\ \Result\,A \to \SLang\,A\)
  satisfying simultaneously: (i) \(\ok\,v \mapsto\) point mass; (ii) bind
  distributes as \(\mathrm{probBind}\); (iii) \fillbytes{} draws a big-endian nat
  uniformly (\emph{the only stochastic claim}); (iii.b) \fillbytes{} preserves
  buffer length; (iv) an Aeneas \texttt{loop} distributes as SampCert
  \(\probWhile\). The \emph{single} assumption about randomness in the crate.
\end{theorem}
\begin{proof}\leanok Declared as an \texttt{axiom}. Trusted. \end{proof}

\begin{definition}[The semantics functor]\label{def:samplerDistGen}
  \lean{OpenDP.Core.Semantics.samplerDistGen}\leanok
  \uses{ax:sdg-exists}
  \(\sdg\,(\mathit{prog})\) is the canonical \(\SLang\) law of any \(\Result\,A\)
  program, chosen from \ref{ax:sdg-exists}. \statusDone
\end{definition}

\begin{definition}[Success distribution]\label{def:samplerDist}
  \lean{OpenDP.Core.Semantics.samplerDist}\leanok
  \uses{def:samplerDistGen}
  \(\sd\,(\mathit{prog})\,v\) is the probability \(\mathit{prog}\) returns
  \(\Ok\,v\); sub-probabilistic, the deficit being failure mass. \statusDone
\end{definition}

\begin{lemma}[Bind distributes over \(\sd\)]\label{thm:samplerDist-bind}
  \lean{OpenDP.Core.Semantics.samplerDist_bind}\leanok
  \uses{def:samplerDist}
  Composition of \(\Result\) programs distributes as \(\mathrm{probBind}\) under
  \(\sd\), with \(\Err\) carrying zero mass. \statusDone
\end{lemma}
\begin{proof}\leanok
  From axiom constraints (i),(ii) plus \texttt{tsum\_result\_ok\_eq}.
\end{proof}

\begin{lemma}[Master \fillbytes{} bridge]\label{thm:fill-bytes-bridge}
  \lean{OpenDP.Core.Semantics.fill_bytes_nat_bridge}\leanok
  \uses{def:samplerDistGen, def:uniformByteNatPMF}
  Any nat-indexed weight summed against the body's first \fillbytes{} draw
  collapses to \(\mathrm{uniformByteNatPMF}\) weighted by that function. Every
  sampler's randomness flows through here. \statusDone
\end{lemma}
\begin{proof}\leanok Routes through axiom constraint (iii). \end{proof}

\begin{lemma}[Loops are \(\probWhile\)]\label{thm:samplerDist-loop}
  \lean{OpenDP.Core.Semantics.samplerDistGen_loop}\leanok
  \uses{def:samplerDistGen}
  An Aeneas \texttt{loop} distributes as \(\probWhile\) of the body's
  distribution, lifting state through \(\mathrm{ControlFlow}\). \statusDone
\end{lemma}
\begin{proof}\leanok From axiom constraint (iv). \end{proof}

  \lean{OpenDP.samplers.uniform.uniformNatBelowPMF}\leanok

\begin{theorem}[Setup totality]\label{ax:uniform-setup}
  \lean{OpenDP.samplers.uniform.sample_uniform_ubig_below_setup_exists}\leanok
  \statusAxiom{} For positive \texttt{upper} the deterministic setup phase
  (bit-length \(\to\) byte-length \(\to\) rejection threshold) succeeds. \statusDone
\end{theorem}
\begin{proof}\leanok Declared as an \texttt{axiom}. \end{proof}

\begin{lemma}[Loop-body reshaping past \texttt{deref\_mut}]\label{thm:body-eq}
  \lean{OpenDP.samplers.uniform.body_eq}\leanok
  \uses{thm:samplerDist-loop}
  The a14083a6 backend wraps the loop body's \fillbytes{} draw in a
  \texttt{deref\_mut} back-function. \texttt{body\_eq} restates the body in clean
  \(\gg\!=\) form (single binder + projections), proved \texttt{by unfold; rfl};
  \texttt{simp\_rw [body\_eq]} then exposes the \fillbytes{} bind so the standard
  bridge applies. This is the reusable key to every a14083a6 loop proof.
  \statusDone
\end{lemma}
\begin{proof}\leanok Definitional (\texttt{unfold; rfl}). \end{proof}

\begin{lemma}[Accepted-outcome mass]\label{thm:body-done}
  \lean{OpenDP.samplers.uniform.body_done_nat_dist}\leanok
  \uses{thm:body-eq, thm:fill-bytes-bridge}
  The \(\done(\Ok\,\cdot)\) mass of one loop step equals
  \(\sum_k \mathrm{uniformByteNatPMF}\,[k<\mathit{thr}\wedge k \bmod
  \mathit{upper}=v]\). \statusDone
\end{lemma}
\begin{proof}\leanok \texttt{body\_eq} entry, then the legacy body analysis. \end{proof}

\begin{lemma}[Rejected-outcome mass]\label{thm:body-cont}
  \lean{OpenDP.samplers.uniform.body_cont_total_mass}\leanok
  \uses{thm:body-eq, thm:fill-bytes-bridge}
  The total \(\cont\) (reject) mass of one loop step equals \(\sum_k
  \mathrm{uniformByteNatPMF}\,[k \ge \mathit{thr}]\). \statusDone
\end{lemma}
\begin{proof}\leanok As \ref{thm:body-done}; injective-\(\cont\) point mass. \end{proof}

\begin{lemma}[Rejection loop is uniform]\label{thm:uniform-pmf}
  \lean{OpenDP.samplers.uniform.sample_uniform_ubig_below_pmf}\leanok
  \uses{thm:body-done, thm:body-cont, thm:samplerDist-loop, def:uniformNatBelowPMF}
  The extracted rejection loop distributes as
  \(\probUntil(\mathrm{uniformByteNatPMF},\ \cdot<\mathit{thr})\), and after
  rejection the mapped output is uniform on \([0,\mathit{upper})\). \statusDone
\end{lemma}
\begin{proof}\leanok
  Splits the body into done/continue fibres (\ref{thm:body-done},
  \ref{thm:body-cont}); assembles via the loop theorem.
\end{proof}

\begin{theorem}[End-to-end: \texttt{sample\_uniform\_ubig\_below}]\label{thm:uniform-spec}
  \lean{OpenDP.samplers.uniform.sample_uniform_ubig_below_spec}\leanok
  \uses{thm:uniform-pmf, ax:uniform-setup}
  For any positive \texttt{upper}, the extracted function has a setup witness,
  reduces to the rejection loop, and its success distribution is exactly
  \(\Unif{\mathit{upper}}\). \statusDone
\end{theorem}
\begin{proof}\leanok From \ref{thm:uniform-pmf} and \ref{ax:uniform-setup}. \end{proof}

\chapter{Bernoulli (rational)}\label{chap:bernoulli}
% =============================================================================

Files: \texttt{src/samplers/bernoulli/\{rational,pmf\}.lean}. The simplest
end-to-end distributional proof, and the ``Lean that reads like the prose''
target. Human proof: sample uniformly from \([0,\denom)\); return \(\top\) iff
below \(\numer\); so \(\Pr[\top]=\numer/\denom\).

\begin{definition}[Bernoulli law as mapped uniform]\label{def:bernoulliPMF}
  \lean{OpenDP.samplers.bernoulli.bernoulliPMF}\leanok
  \uses{def:uniformNatBelowPMF}
  \(\mathrm{bernoulliPMF}\,\mathit{num}\,\mathit{den} = \PMF.\mathrm{map}\,(x
  \mapsto x<\mathit{num})\,(\mathrm{UniformSample}\,\mathit{den})\). \statusDone
\end{definition}

\begin{lemma}[Matches SampCert's Bernoulli]\label{thm:bernoulliPMF-eq}
  \lean{OpenDP.samplers.bernoulli.bernoulliPMF_eq_BernoulliSamplePMF}\leanok
  \uses{def:bernoulliPMF}
  \(\mathrm{bernoulliPMF} = \SLang.\mathrm{BernoulliSamplePMF}\); grounds the
  result against the trusted reference. \statusDone
\end{lemma}
\begin{proof}\leanok Pointwise on \(\{\top,\bot\}\) via SampCert apply lemmas. \end{proof}

\begin{theorem}[Translation layer]\label{thm:bernoulli-eq-setup}
  \lean{OpenDP.samplers.bernoulli.sample_bernoulli_rational_eq_of_setup}\leanok
  \uses{thm:uniform-spec}
  On the valid-input branch the extracted function reduces to a uniform draw
  below \(\denom\) followed by the comparison closure \(\numer>\cdot\). \statusDone
\end{theorem}
\begin{proof}\leanok \texttt{unfold} + the setup specs. \end{proof}

\begin{theorem}[End-to-end: \texttt{sample\_bernoulli\_rational}]\label{thm:bernoulli-rational}
  \lean{OpenDP.samplers.bernoulli.sample_bernoulli_rational_pmf}\leanok
  \uses{thm:bernoulli-eq-setup, thm:uniform-pmf, def:bernoulliPMF, thm:bernoulliPMF-eq}
  \(\sd(\text{extracted sample\_bernoulli\_rational}) = \mathrm{bernoulliPMF}\),
  i.e.\ SampCert's \(\Bern{\numer/\denom}\). \statusDone
\end{theorem}
\begin{proof}\leanok
  Push \(\sd\) through the comparison closure (\texttt{Result.map}) and the
  uniform PMF via a \(\mathrm{UBig}\!\to\!\mathrm{Nat}\) bridge.
\end{proof}

\chapter{Roadmap: toward the discrete Gaussian}\label{chap:roadmap}
% =============================================================================

The remaining chain (see \texttt{ROADMAP.md}) continues linearly from the rational
Bernoulli up to the discrete Gaussian noise mechanism. Each node below is one
stage; the \texttt{uses} edges are the pipeline. They start as stubs and each turns
green the moment its \(\sd(\cdot) = \text{SampCert PMF}\) theorem lands in
\texttt{repro/src/} --- so this graph \emph{is} the progress bar. Marked
\statusStub{} = target only: (extracted, not ported) extracted (Rust in \texttt{Generated/}, proof not
yet ported) or (not extracted) not extracted (needs a Charon/Aeneas pass first).

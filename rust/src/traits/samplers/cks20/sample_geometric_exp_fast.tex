Proves soundness of \texttt{fn sample\_geometric\_exp\_fast} in \asOfCommit{mod.rs}{0be3ab3e6}. 
This proof is an adaptation of \href{https://arxiv.org/pdf/2004.00010.pdf#subsection.5.2}{subsection 5.2} of \cite{CKS20}. 
 
\section{Hoare Triple} 
 
\subsection*{Precondition} 
\subsubsection*{Compiler-verified} 
Argument \texttt{x} is of type \texttt{RBig}, a bignum rational. 
 
\subsubsection*{Caller-verified} 
$\texttt{x} \geq 0$ 
 
\subsection*{Pseudocode} 
\lstinputlisting[language=Python,firstline=2,escapechar=|]{./pseudocode/sample_geometric_exp_fast.py} 
 
\subsection*{Postcondition} 
\label{postcondition} 
For any setting of the input parameter \texttt{x} such that the given preconditions hold, \\ 
\texttt{sample\_geometric\_exp\_fast} either returns \texttt{Err(e)} due to a lack of system entropy, 
or \texttt{Ok(out)}, where \texttt{out} is distributed as $\textrm{Geometric}(1 - \exp(-x))$. 
 
\section{Proof} 
Assume the preconditions are met. 
 
\begin{lemma}\label{err-e} 
    \texttt{sample\_geometric\_exp\_fast} only returns \texttt{Err(e)} when there is a lack of system entropy. 
\end{lemma} 
 
\begin{proof} 
    $\texttt{x} \geq 0$ and \texttt{x} is of type \docsrs{dashu}{rational/struct}{Rbig}, 
    so there exists some non-negative integer $s$ and positive integer $t$ such that $x = s/t$. 
    This is why \texttt{Rational.into\_numer\_denom} is infallible. 
    Since $t$ is a positive integer, the preconditions on \rustdoc{traits/samplers/uniform/fn}{sample\_uniform\_ubig\_below} are met, 
    \texttt{sample\_uniform\_int\_below} can only return an error due to lack of system entropy, and $u$ is a non-negative integer. 
    Similarly, the preconditions on \rustdoc{traits/samplers/cks20/fn}{sample\_bernoulli\_exp} and \rustdoc{traits/samplers/cks20/fn}{sample\_geometric\_exp\_slow} are met, 
    and their definitions guarantee an error is only returned due to lack of system entropy. 
    The only source of errors is from the invocation of these functions, 
    therefore \texttt{sample\_geometric\_exp\_fast} only returns \texttt{Err(e)} when there is a lack of system entropy. 
\end{proof} 
 
We now establish some lemmas that will be useful in proving the distribution of \texttt{out}. 
 
\begin{lemma}\label{udv-dists} 
    Conditioned on \texttt{sample\_geometric\_exp\_fast} not returning \texttt{Err(e)}, the following hold: 
    \begin{itemize} 
        \item The value \texttt{u} sampled on line \ref{line:U} is a realization of 
              $U \sim \textrm{Uniform}(0,t)$ supported on $\{0,1,\dots,t-1\}$. 
        \item Conditioned on a fixed realization $u$ from line \ref{line:U}, the value \texttt{d} sampled on line \ref{line:D} 
              is a realization of $D \sim \textrm{Bernoulli}(\exp(-u/t))$. 
        \item The value \texttt{v} sampled on line \ref{line:V} is a realization of 
              $V \sim \textrm{Geometric}(1-\exp(-1))$. 
        \item Moreover, $U$ is independent of $V$, and the call on line \ref{line:V} is independent of the loop outcome on lines \ref{line:U}--\ref{line:D}. 
    \end{itemize} 
\end{lemma} 
 
\begin{proof} 
    Since \texttt{x} is an \texttt{RBig} rational and $\texttt{x}\ge 0$, there exist integers $s\ge 0$ and $t>0$ with $x=s/t$, 
    so the argument $t$ passed to \rustdoc{traits/samplers/uniform/fn}{sample\_uniform\_ubig\_below} on line \ref{line:U} satisfies its preconditions. 
    By the specification/definition of \rustdoc{traits/samplers/uniform/fn}{sample\_uniform\_ubig\_below}, 
    the returned value \texttt{u} is uniformly distributed over $\{0,\dots,t-1\}$, i.e., $U\sim \textrm{Uniform}(0,t)$. 
 
    On line \ref{line:D}, the call \rustdoc{traits/samplers/cks20/fn}{sample\_bernoulli\_exp} is made with input \texttt{Rational(u,t)}. 
    The preconditions of \rustdoc{traits/samplers/cks20/fn}{sample\_bernoulli\_exp} are met (in particular, $u\in[0,t)$ and $t>0$), 
    and by its specification/definition, conditioned on the realized \texttt{u}, 
    the returned bit \texttt{d} is distributed as $\textrm{Bernoulli}(\exp(-u/t))$. 
 
    On line \ref{line:V}, the call \rustdoc{traits/samplers/cks20/fn}{sample\_geometric\_exp\_slow} with argument $1$ 
    returns \texttt{v} distributed as $\textrm{Geometric}(1-\exp(-1))$ by its specification/definition. 
 
    Independence follows because line \ref{line:V} is executed after the loop and uses fresh randomness. 
\end{proof} 
 
\begin{lemma}\label{geom_1_t}\cite{CKS20} 
    Assume \texttt{u}, \texttt{d}, and \texttt{v} satisfy the conditions of Lemma \ref{udv-dists}. 
    Conditioned on $d = \top$, if $z = u + t \cdot v$, 
    then $z$ is a realization of a random variable $Z \sim \textrm{Geometric}(1 - \exp(-1/t))$. 
    Equivalently, $P[Z=z \mid D=\top] = (1 - e^{-1/t}) e^{-z/t}$. 
\end{lemma} 
 
\begin{proof} 
    For any $z$, define $u_z := z \mod t$ and $v_z := \lfloor z/t \rfloor$, so that $z = u_z + t \cdot v_z$. 
 
    \begin{align*} 
        P[Z=z \mid D=\top] 
        &= P[U = u_z, V = v_z \mid D = \top] && \text{since } z = u_z + t \cdot v_z \\ 
        &= P[U = u_z \mid D = \top] \, P[V = v_z] && \text{as $U$ and $V$ are independent, and $V$ and $D$ are independent} \\ 
        &= \frac{P[U = u_z]}{P[D=\top]} P[D=\top \mid U = u_z] \cdot (1 - e^{-1}) e^{-v_z} && \text{by Bayes' theorem} \\ 
        &= \frac{1/t}{1/t \sum_{k=0}^{t-1}e^{-k/t}} e^{-u_z/t} \cdot (1 - e^{-1}) e^{-v_z} 
        && \text{since } P[D = \top] = \frac{1}{t} \sum_{k=0}^{t-1}e^{-k/t} \\ 
        &= \frac{(1 - e^{-1})}{\sum_{k=0}^{t-1}e^{-k/t}}  e^{-(u_z/t + v_z)} \\ 
        &= \frac{(1 - e^{-1})}{(1 - (e^{-1/t})^t)/(1 - e^{-1/t})} \\ 
        &= (1 - e^{-1/t}) e^{-(u_z/t + v_z)} \\ 
        &= (1 - e^{-1/t}) e^{-z/t} && \text{since } z = u_z + t \cdot v_z 
    \end{align*} 
\end{proof} 
 
\begin{lemma}\label{divide_geometric}\cite{CKS20} 
    Fix $p \in (0, 1]$. Let $G$ be a $\textrm{Geometric}(1 - p)$ random variable, and $n \geq 1$ be an integer. 
    Then $\lfloor G / n \rfloor$ is a $\textrm{Geometric}(1 - q)$ random variable with $q = p^n$. 
\end{lemma} 
 
\begin{proof} 
    \begin{align*} 
        P[\lfloor G/n \rfloor = k] &= P[nk \le G < (k + 1)n] &&\text{any $G$ in the interval maps to $k$} \\ 
        &= \sum_{l=kn}^{(k+1)n - 1} (1 - p)p^l \\ 
        &= (1 - p^n)p^{nk} \\ 
        &= (1 - q)q^k 
    \end{align*} 
\end{proof} 
 
\begin{lemma}\label{ok-out} 
    If the outcome of \texttt{sample\_geometric\_exp\_fast} is \texttt{Ok(out)}, 
    then \texttt{out} is distributed as $\textrm{Geometric}(1 - \exp(-x))$. 
\end{lemma} 
 
\begin{proof} 
    By Lemma \ref{udv-dists}, \texttt{u}, \texttt{d}, and \texttt{v} have the distributions (and independence) 
    required to apply Lemma \ref{geom_1_t}. 
    By Lemma \ref{geom_1_t}, \texttt{z} is a realization of $Z \sim \textrm{Geometric}(1 - \exp(-1/t))$; hence by Lemma \ref{divide_geometric}, 
    $\lfloor Z / s \rfloor$ is distributed as $\textrm{Geometric}(1 - \exp(-s/t))$. 
    Finally, $\texttt{out} = \lfloor Z / s \rfloor$ and $s/t = x$, 
    so \texttt{out} is distributed as $\textrm{Geometric}(1 - \exp(-x))$. 
\end{proof} 
 
\begin{proof} 
    \ref{postcondition} holds by \ref{err-e} and \ref{ok-out}. 
\end{proof}

\subsection*{Postcondition} 
\begin{definition} 
    Define $X \sim \mathcal{L}_\mathbb{Z}(0, s)$, a random variable following the discrete laplace distribution: 
    \begin{equation} 
        \forall x \in \mathbb{Z} \qquad \Pr[X = x] = \frac{1 - e^{-1/s}}{1 + e^{-1/s}} e^{-|x|/s} 
    \end{equation} 
\end{definition} 
 
\begin{theorem} 
    Assume $X \sim \mathcal{L}_\mathbb{Z}(0, s)$, and $t > 0$, where $s = \texttt{scale}$ and $t = \texttt{tail}$. 
    \begin{equation} 
        \alpha = P[X > t] = \frac{e^{-t/s}}{e^{1/s} + 1} 
    \end{equation} 
\end{theorem} 
 
\begin{proof} 
    \begin{align*} 
        \alpha &= P[X > t] \\ 
        &= \sum_{x=t + 1}^{\infty} \frac{1 - e^{-1/s}}{1 + e^{-1/s}} e^{-|x|/s} \\ 
        &= \frac{1 - e^{-1/s}}{1 + e^{-1/s}} \sum_{x=t + 1}^{\infty} e^{-x/s}  
            && \text{since } t > 0 \\ 
        &= \frac{1 - e^{-1/s}}{1 + e^{-1/s}} \frac{e^{-(t + 1)/s}}{1 - e^{-1/s}}  
            && \text{since } \sum_{x=t}^\infty p^x = \frac{p^t}{1 - p} \text{ if } |p| < 1 \text{, let } p = e^{-1/s}\\ 
        &= \frac{e^{-(t + 1)/s}}{1 + e^{-1/s}} \\ 
        &= \frac{e^{-t/s}}{e^{1/s} + 1} 
    \end{align*} 
 
\end{proof}

Proof for \rustdoc{accuracy/tail\_bounds/fn}{integrate\_discrete\_laplace\_tail}.

\begin{definition}
    Define $Y \sim \mathcal{L}_\mathbb{Z}(0, s)$, a random variable following the discrete laplace distribution:
    \begin{equation}
        \forall x \in \mathbb{Z} \qquad \Pr[X = x] = \frac{e^{1/s} - 1}{e^{1/s} + 1} e^{-|x|/s}
    \end{equation}
\end{definition}

\begin{theorem}
    Assume $X \sim \mathcal{L}_\mathbb{Z}(0, s)$.
    \begin{equation}
        \alpha = P[X \ge t] = \frac{e^{-t/s}}{1 + e^{-1/s}}
    \end{equation}
\end{theorem}

\begin{proof}
    First, assume $t > 0$.
    \begin{align*}
        \alpha &= P[X \ge t] \\
        &= \sum_{x=t}^{\infty} \frac{e^{1/s} - 1}{e^{1/s} + 1} e^{-|x|/s} \\
        &= \frac{e^{1/s} - 1}{e^{1/s} + 1} \sum_{x=t}^{\infty} e^{-x/s} 
            && \text{since } t > 0 \\
        &= \frac{e^{1/s} - 1}{e^{1/s} + 1} \frac{e^{(1-t)/s}}{e^{1/s} - 1} 
            && \text{since } \sum_{x=t}^\infty p^x = \frac{p^x}{1 - p} \text{ if } |p| < 1 \text{, let } p = e^{1/s}\\
        &= \frac{e^{(1 - t)/s}}{e^{1/s} + 1} \\
        &= \frac{e^{-t/s}}{1 + e^{-1/s}} && \text{for numerical stability}
    \end{align*}

\end{proof}
